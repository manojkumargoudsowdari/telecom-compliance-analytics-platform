# Section: raw page verification
@'
import json
from pathlib import Path

metadata_root = Path('data/raw/fcc/_metadata')
metadata_files = [p for p in metadata_root.glob('*.json') if p.name != '.gitkeep']
latest_metadata = max(metadata_files, key=lambda p: p.stat().st_mtime)
metadata = json.loads(latest_metadata.read_text(encoding='utf-8'))
run_id = metadata['run_id']
run_directory = Path(metadata['landing_directory'])
page_files = sorted(run_directory.glob('*.json'))

all_arrays = True
all_objects = True
empty_pages = []
record_total = 0
for path in page_files:
    payload = json.loads(path.read_text(encoding='utf-8'))
    is_array = isinstance(payload, list)
    is_object_array = is_array and all(isinstance(item, dict) for item in payload)
    if not is_array:
        all_arrays = False
    if not is_object_array:
        all_objects = False
    if is_array:
        record_total += len(payload)
        if len(payload) == 0:
            empty_pages.append(path.name)

last_page = page_files[-1]
last_payload = json.loads(last_page.read_text(encoding='utf-8'))
expected_next_page = run_directory / f'consumer_complaints_page_{len(page_files) + 1:06d}.json'

lines = []
lines.append('COMMAND: raw page verification from latest successful FCC raw ingestion run')
lines.append(f'RUN_ID: {run_id}')
lines.append(f'RUN_DIRECTORY: {run_directory.as_posix()}')
lines.append(f'PAGE_FILE_COUNT: {len(page_files)}')
lines.append(f'FIRST_PAGE: {page_files[0].name}')
lines.append(f'LAST_PAGE: {last_page.name}')
lines.append(f'ALL_FILES_ARE_JSON_ARRAYS: {all_arrays}')
lines.append(f'ALL_ARRAY_ITEMS_ARE_OBJECTS: {all_objects}')
lines.append(f'EMPTY_PAGE_FILES: {empty_pages if empty_pages else "NONE"}')
lines.append(f'LAST_PAGE_RECORD_COUNT: {len(last_payload)}')
lines.append(f'NO_TERMINAL_EMPTY_PAGE_FILE: {len(last_payload) > 0 and not expected_next_page.exists() and not empty_pages}')
lines.append(f'TOTAL_RECORDS_FROM_FILES: {record_total}')
lines.append(f'RECORD_COUNTS_RECONCILE: {record_total == metadata.get("total_records_fetched")}')
lines.append(f'EXPECTED_NEXT_PAGE_PRESENT: {expected_next_page.exists()}')
lines.append('')
lines.append('SAMPLE_FIRST_PAGE_OBJECT:')
lines.append(json.dumps(json.loads(page_files[0].read_text(encoding="utf-8"))[0], indent=2))
print('\n'.join(lines))
